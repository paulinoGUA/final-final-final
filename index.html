<script>

  // Variables globales para guardar palabras originales (para resetear)

  let juego1Palabras = Array.from(document.querySelectorAll('#grupo-palabras .draggable')).map(el => el.outerHTML);

  let juego2Palabras = Array.from(document.querySelectorAll('#palabras-juego2 .draggable')).map(el => el.outerHTML);


  // Drag & Drop handlers

  function allowDrop(ev) {

    ev.preventDefault();

    if(ev.target.classList.contains('dropzone')){

      ev.target.classList.add('dragover');

    }

  }

  function drag(ev) {

    ev.dataTransfer.setData("text", ev.target.innerText);

  }

  function drop(ev, zona) {

    ev.preventDefault();

    if(ev.target.classList.contains('dropzone')){

      ev.target.classList.remove('dragover');

      // Vaciar contenido previo y agregar el texto arrastrado

      ev.target.innerHTML = '';

      let texto = ev.dataTransfer.getData("text");

      // Crear nuevo span draggable pero sin permitir volver a arrastrar dentro del dropzone (para evitar bugs)

      let span = document.createElement('span');

      span.textContent = texto;

      span.className = 'draggable';

      ev.target.appendChild(span);

    }

  }

  function dragLeave(ev) {

    if(ev.target.classList.contains('dropzone')){

      ev.target.classList.remove('dragover');

    }

  }


  // Juego 1: Verificar respuestas

  function verificarJuego1() {

    const mamaZone = document.querySelectorAll('#juego1 .dropzone')[0];

    const wawaZone = document.querySelectorAll('#juego1 .dropzone')[1];

    const mamaCorrectas = ["Cushma", "Chumpi", "Sombrero blanco", "Pinkullu", "Bombo", "Huactana"];

    const wawaCorrectas = ["Pollerín", "Alfanje", "Cascabeles", "Chanta", "Bandas"];


    let mamaUser = mamaZone.textContent.trim().split(/\s+/);

    let wawaUser = wawaZone.textContent.trim().split(/\s+/);


    // Validar que palabras del usuario estén en la lista correcta, sin palabras que no correspondan

    const mamaOk = mamaUser.every(p => mamaCorrectas.includes(p)) && mamaUser.length > 0;

    const wawaOk = wawaUser.every(p => wawaCorrectas.includes(p)) && wawaUser.length > 0;


    let retro = document.getElementById('retro-juego1');

    if(mamaOk && wawaOk){

      retro.style.color = '#2e7d32';

      retro.textContent = "¡Excelente! Has clasificado correctamente las palabras del Mama Danza y Wawa Danza.";

    } else {

      retro.style.color = '#c62828';

      retro.textContent = "Revisa las palabras y vuelve a intentarlo. ¡Cada error es una oportunidad para aprender!";

    }

  }

  // Juego 1: Resetear

  function resetJuego1() {

    // Limpiar dropzones

    const mamaZone = document.querySelectorAll('#juego1 .dropzone')[0];

    const wawaZone = document.querySelectorAll('#juego1 .dropzone')[1];

    mamaZone.innerHTML = '';

    wawaZone.innerHTML = '';

    // Restaurar palabras al grupo

    const grupo = document.getElementById('grupo-palabras');

    grupo.innerHTML = juego1Palabras.join('');

    document.getElementById('retro-juego1').textContent = '';

  }


  // Juego 2: Verificar

  function verificarJuego2() {

    const zonas = document.querySelectorAll('#juego2 .dropzone');

    const respuestas = {

      1: "Inti",

      2: "Atuendo",

      3: "Alfanje",

      4: "Tauna"

    };

    let ok = true;

    for(let i=0; i<zonas.length; i++){

      const val = zonas[i].textContent.trim();

      if(val !== respuestas[i+1]){

        ok = false;

        break;

      }

    }

    const retro = document.getElementById('retro-juego2');

    if(ok){

      retro.style.color = '#2e7d32';

      retro.textContent = "¡Muy bien! Cada palabra honra nuestras raíces y cultura andina.";

    } else {

      retro.style.color = '#c62828';

      retro.textContent = "Revisa tus respuestas y vuelve a intentarlo. ¡Aprender es un camino!";

    }

  }

  // Juego 2: Resetear

  function resetJuego2() {

    const zonas = document.querySelectorAll('#juego2 .dropzone');

    zonas.forEach(z => z.innerHTML = '');

    const grupo = document.getElementById('palabras-juego2');

    grupo.innerHTML = juego2Palabras.join('');

    document.getElementById('retro-juego2').textContent = '';

  }
